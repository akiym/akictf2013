function argmin(array $values)
{
    $argmin = $min = null;
    foreach ($values as $key => $value) {
        if ($min === null || $min > $value) {
            $argmin = $key;
            $min = $value;
        }
    }
    return $argmin;

import urllib.request
import urllib.parse
import os
import re
from PIL import Image
import io
import pyocr
import pyocr.builders

datas = []

TEMP_DIR = "./tmp/"
if not os.path.isdir(TEMP_DIR): os.makedirs(TEMP_DIR)

def getimg(url):
    try:
        resp = urllib.request.urlopen(url)
    except urllib.error.HTTPError:
        return None

    data = resp.readall()

    datas.append(data)
    return datas

def geturl(str):
    m = re.compile("/challenge/.*\.png")
    m = m.search(str)
    return "http://q34.ctf.katsudon.org" + m.group(0)

def mgImg(datas, CN=0):
    img = Image.open(io.BytesIO(datas[0]))
    maxx = img.size[0]
    maxy = img.size[1]
    pix = img.load()

    for d in datas[1:]:
        im = Image.open(io.BytesIO(d))
        pi = im.load()

        for x in range(maxx):
            for y in range(maxy):
                if pix[x, y] == 0:
                    continue
                if pi[x, y] != pix[x, y]:
                    pix[x, y] = 0xffffff

    img.save("./tmp/"+ str(CN) +".png", "png")
    return img

def ocr(img):
    tools = pyocr.get_available_tools()
    tool = tools[0]
    txt = tool.image_to_string(img, lang="eng", builder=pyocr.builders.TextBuilder())
    print("key:" + txt)
    return txt
